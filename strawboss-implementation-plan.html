// apps/mobile/src/components/shared/RoleHeader.tsx
import { LinearGradient } from 'expo-linear-gradient'
import { gradients, roleColors } from '@strawboss/ui-tokens'

interface RoleHeaderProps {
  role: 'driver' | 'baler' | 'loader'
  name: string
  machineCode?: string
  subtitle?: string
  isOnline: boolean
}

export function RoleHeader({ role, name, machineCode, isOnline }: RoleHeaderProps) {
  const grad = gradients[role]
  const ROLE_LABELS = { driver: 'ȘOFER', baler: 'BALOTIERĂ', loader: 'ÎNCĂRCĂTOR' }

  return (
    <LinearGradient colors={grad} style={styles.header}>
      {/* Sync dot */}
      <View style={styles.syncDot(isOnline)} />
      {/* Machine chip */}
      {machineCode && <View style={styles.machineChip}><Text>{machineCode}</Text></View>}
      <Text style={styles.greeting}>Bună ziua,</Text>
      <Text style={styles.name}>{name}</Text>
      <View style={styles.badge}>
        <Text style={styles.badgeText}>{ROLE_LABELS[role]}</Text>
      </View>
    </LinearGradient>
  )
}
